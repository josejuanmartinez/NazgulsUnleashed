blind_sorcerer_monthly_check = {
	trigger = {
		root = character:nazgul5
		OR = {
			AND = {
				is_sorcerer = yes
				NOT = { has_character_modifier = practiced_sorcery_recently }
			}
			any_courtier = {
				is_sorcerer = yes
				NOT = {
					has_character_modifier = practiced_sorcery_recently
				}
			}	
		}
		gold > 0
		piety > 0
	}

	random_events = {
		chance_to_happen = 25
		chance_of_no_event = {
			if = {
				limit = { has_game_rule = very_easy_difficulty }
				value = 66
			}
			else_if = {
				limit = { has_game_rule = easy_difficulty }
				value = 33
			}
		}
		
	}
}
