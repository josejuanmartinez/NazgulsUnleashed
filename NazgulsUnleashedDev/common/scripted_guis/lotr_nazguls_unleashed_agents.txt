can_do_schemes_indur = {
	is_shown = {
		root = character:nazgul4
	}
	is_valid = {
		OR = {
			AND = {
				has_trait = indur_scheme_court			
				NOT = {
					has_character_modifier = schemed_recently
				}
			}			
			any_courtier = {
				has_trait = indur_scheme_court
				NOT = {
					has_character_modifier = schemed_recently
				}
			}	
		}
	}
}

unselect_agent = {	
	effect = {
		remove_variable = selected_agent
		remove_variable = selected_scheme
	}
}

unselect_scheme = {	
	effect = {
		remove_variable = selected_scheme
	}
}

select_agent = {
	effect = {
		trigger_event = agent_schemes.0001
	}
}

select_steal_gold = {
	is_valid = {
		always = yes
	}
	effect = {
		set_variable = {
			name = selected_scheme
			value = flag:steal_gold
		}
	}
}

trigger_steal_gold = {
	effect = 	{
		var:selected_agent = { add_character_modifier = {
			modifier = schemed_recently
			years = 4
		} }

		trigger_event = agent_schemes.0010

	}
}
